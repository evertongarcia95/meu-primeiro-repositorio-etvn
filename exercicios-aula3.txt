// Exemplo 01 - IF

	funcao inicio() {
		
		const cadeia diaSemana = "7"

		se (diaSemana == "1")

			escreva("\nHoje é Domingo\n")

		{
			escreva("\nHoje é Segunda-feira\n") 
		}	
		senao se(diaSemana == "3")
		{
			escreva("\nHoje é Terça-feira\n") 
		}	
		senao se(diaSemana == "4")
		{
			escreva("\nHoje é Quarta-feira\n") 
		}	
		senao se(diaSemana == "5")
		{
			escreva("\nHoje é Quinta-feira\n") 
		}	
		senao se(diaSemana == "6")
		{

		{
			escreva("\nHoje é Sabado\n") 
		}	
	}
}

// Exemplo 03 - FOR
programa 
{
	funcao inicio() 
	{
		inteiro numero, resultado, contador
		
		escreva("Informe um número para ver sua tabuada: ")
		leia(numero)

		limpa()
		
		para (contador = 1; contador <= 10; contador++) 
		{
			resultado = numero * contador 
			escreva (numero, " X ", contador, " = ", resultado , "\n")
		}
	}
}
